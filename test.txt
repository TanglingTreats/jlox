class Cake {
  init() {
    this.flavor = "Raspberry";
  }
  taste() {
    var adjective = "delicious";
    print "The " + this.flavor + " cake is " + adjective + "!";
  }
}

var cake = Cake();
cake.taste();

cake.flavor = "German chocolate";
cake.taste(); // Prints "The German chocolate cake is delicious!".
